# ffmpeg

```ts
import ffmpeg from 'fluent-ffmpeg';
import ffmpegPath from '@ffmpeg-installer/ffmpeg';
import ffprobePath from '@ffprobe-installer/ffprobe';
ffmpeg.setFfmpegPath(ffmpegPath.path);
ffmpeg.setFfprobePath(ffmpegPath.path);

//识别编码类型
function getVideoCodec(filePath: string) {
  return new Promise((resolve, reject) => {
    ffmpeg.ffprobe(filePath, (err, metadata) => {
      if (err) return reject(err);
      // 找到第一个视频流
      const videoStream = metadata.streams.find((stream) => stream.codec_type === 'video');
      if (videoStream && videoStream.codec_name) {
        resolve(videoStream.codec_name); // codec_name如："h264" 或 "hevc"（hevc就是h265）
      } else {
        reject(new Error('No video stream found'));
      }
    });
  });
}
```

查询视频信息

```ts
export const getVideoInfo = (inputPath: string) => {
  return new Promise((resolve, reject) => {
    const args = [
      '-v',
      'quiet', // 设置日志级别为静默，只输出 -show_... 请求的数据
      '-print_format',
      'json', // 输出格式为 JSON
      '-show_format', // 显示格式信息 (如总时长、文件大小、格式名称等)
      '-show_streams', // 显示所有流的信息 (视频流、音频流、字幕流等)
      inputPath // 输入文件路径
    ];

    const ffprobeProcess = spawn(ffprobePath, args);

    let outputData = ''; // 用于收集子进程的输出

    // 监听 ffprobe 的标准输出 (stdout)
    ffprobeProcess.stdout.on('data', (chunk) => {
      outputData += chunk; // 将接收到的数据块拼接到 outputData 字符串中
    });

    // 监听 ffprobe 的标准错误输出 (stderr)
    // 如果 ffprobe 遇到错误（如文件不存在、格式不支持），错误信息会在这里输出
    ffprobeProcess.stderr.on('data', (errorChunk) => {
      console.error(`FFprobe stderr: ${errorChunk}`);
    });

    // 监听 ffprobe 进程的 'close' 事件，当进程结束时触发
    ffprobeProcess.on('close', (code) => {
      if (code === 0) {
        // 如果进程正常退出 (code === 0)，则尝试解析收集到的 JSON 数据
        try {
          const parsedInfo = JSON.parse(outputData);
          resolve(parsedInfo); // 将解析后的信息对象作为 Promise 的值
        } catch (parseError: any) {
          // 如果输出不是有效的 JSON，说明可能出错了
          console.error('解析 ffprobe 输出失败:', parseError);
          console.error('原始输出:', outputData);
          reject(new Error(`ffprobe 输出无法解析为 JSON: ${parseError.message}`));
        }
      } else {
        // 如果进程异常退出 (code !== 0)，说明 ffprobe 执行失败
        reject(new Error(`ffprobe 进程退出码: ${code}. 请检查文件路径或格式是否正确。`));
      }
    });

    // 监听进程启动错误 (e.g., 'ffprobe' command not found)
    ffprobeProcess.on('error', (err: any) => {
      console.error('启动 ffprobe 进程时发生错误:', err.message);
      if (err.code === 'ENOENT') {
        reject(
          new Error(
            "系统找不到 'ffprobe' 命令。请确保 FFmpeg/ffprobe 已正确安装并添加到系统 PATH 环境变量中。"
          )
        );
      } else {
        reject(err);
      }
    });
  });
};
```

```ts
export const getM3u8 = async (filePath: string, duration1: string, type: string) => {
  const duration = Number(duration1);

  const m3u8List: string[] = [
    '#EXTM3U',
    '#EXT-X-VERSION:3',
    '#EXT-X-TARGETDURATION:10',
    '#EXT-X-MEDIA-SEQUENCE:0'
  ];

  for (let i = 0; i < duration; i += 10) {
    if (i + 10 > duration) {
      m3u8List.push(`#EXTINF:${duration - i},`);
    } else {
      m3u8List.push(`#EXTINF:10,`);
    }
    m3u8List.push(
      `media://video?file=${filePath}&duration=${duration}&type=${type}&start=${Math.floor(i / 10)}`
    );
  }
  m3u8List.push('#EXT-X-ENDLIST');

  return m3u8List.join('\n');
};
```

```ts
const runCMDBuffer = (cmd: string, args: string[]) => {
  return new Promise<Buffer>((resolve, reject) => {
    const proc = spawn(cmd as string, args, {stdio: ['ignore', 'pipe', 'pipe']});
    const chunks: any[] = [];
    proc.stdout.on('data', (d) => {
      chunks.push(d);
    });
    proc.stdout.on('end', () => {
      const buf = Buffer.concat(chunks);
      resolve(buf);
    });

    proc.on('error', reject);
    // proc.stderr.on("data", () => {});
  });
};
const runCMDStr = (cmd: string, args: string[]) => {
  return new Promise<string>((resolve, reject) => {
    const proc = spawn(cmd as string, args, {stdio: ['ignore', 'pipe', 'pipe']});
    let chunks = '';
    proc.stdout.on('data', (d) => {
      chunks += d;
    });
    proc.stdout.on('end', () => {
      resolve(chunks);
    });

    proc.on('error', reject);
    // proc.stderr.on("data", () => {});
  });
};
```
