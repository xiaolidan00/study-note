# 地图适配左右侧栏

```js
onFitView(start, end) {
            this.isAction = true;
            const sideWidth = 340;
            const container = document.getElementById('container');
            const distance = (this.showLeft ? sideWidth : 0) + (this.showRight ? sideWidth : 0);
            const ww = container.offsetWidth - distance;
            const hh = container.offsetHeight;
            const moveCenter = sideWidth * 0.5;
            setTimeout(() => {
                this.isAction = false;
            }, 1000);

            for (let i = 8; i < 18; i++) {
                const p = this.map.project(start, i);
                const p1 = this.map.project(end, i);
                if (Math.abs(p.x - p1.x) >= ww || Math.abs(p.y - p1.y) >= hh) {
                    console.log('fitView', i);
                    const zoom = i - 1;

                    let center = [0.5 * (start[0] + end[0]), 0.5 * (start[1] + end[1])];
                    const p2 = this.map.project(center, zoom);
                    if (this.showLeft && !this.showRight) {
                        center = this.map.unproject([p2.x - moveCenter, p2.y], zoom);
                    } else if (!this.showLeft && this.showRight) {
                        center = this.map.unproject([p2.x + moveCenter, p2.y], zoom);
                    }
                    this.map.setView(center, zoom);
                    console.log('zoom', zoom);
                    this.currentZoom = zoom;
                    return zoom;
                }
            }
        }
```
